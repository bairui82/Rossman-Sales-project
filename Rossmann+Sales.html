def rmspe(real,predict):  
    #Y_Val
    real=list(real)
    #sum_value
    sum_value=0.0  
    # count_num,,Y,,
    count_num=0
    for i in range(len(real)):
        if real[i] == 0:
            continue
        else:
            v1=((real[i]-predict[i])/real[i])**2  
            sum_value += v1
            count_num += 1
    v2=sum_value / count_num
    v3=np.sqrt(v2)
    return v3

def train_predict(model, sample_size, X_train, y_train, X_val, y_val): 
    '''
    inputs:
       - model: 
       - sample_size: sample
       - X_train: 
       - Y_train: 
       - X_val: 
       - Y_val: 
    '''
    
    results = {}
    
    # TODOsample_size
    start = time() # 
    model = model.fit(X_train[:sample_size],y_train[:sample_size])
    end = time() # 
    
    # TODO
    results['train_time'] = end-start
    
    # TODO: 
    start = time() # 
    predictions_val = model.predict(X_val)
    end = time() # 
    
    # TODO
    results['pred_time'] = end-start
               
    # TODOrmspe-score
    results['RMSPE'] = rmspe(Y_val,predictions_val)
       
    # 
    print "{} {} samples.".format(model.__class__.__name__, sample_size)
        
    # 
    return results
